create table head (
	id serial primary key,
	hair_color text,
	face_type text
);

create table body (
	id serial primary key,
	skin_color text,
	finger_amount int,
	arm_amount int
);

create table leg (
	id serial primary key,
	amount text
);

CREATE Table construction (
	id serial primary key,
	head_id serial references head,
	body_id serial references body,
	leg_id serial references leg
);
